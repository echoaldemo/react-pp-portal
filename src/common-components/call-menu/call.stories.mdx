# Call Menu

<Preview>
  <Story name="Card Header">
    {() => {
      const [callInfo, setCallInfo] = useState({
        agentName: "BoyLado Hernandez",
        listening: true,
        whisper: false
      });
      const ToggleCallButton = (button) => {
        setCallInfo({ ...callInfo, [button]: !callInfo[button] });
      };
      return <>
        <CallMenu callInfo={callInfo} ToggleCallButton={ToggleCallButton} />
      </>
    }}
  </Story>
</Preview>

# Props

| Name             | Type     | Default  | Description                    |
| ---------------- | -------- | ---------| ------------------------------ |
| callInfo         | Array    | ---------| required, info of the call     |
| ToggleCallButton | Function | void     | required, onClick Event        |
